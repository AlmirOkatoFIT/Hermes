133 static void morse_nibble( u8 nibble )
134 {
135  u8 i, cod, code_len;
136 
137  if (nibble > 0xf) nibble = 0x89; /* -..- = "x" */
138 
139  cod = codes[nibble];
140  code_len = cod >> 5;
141 
142  for ( i = code_len; i>0; i-- )
143  {
144  if (NUMLEDS == 0) // if there are no status LEDs use the mcu led for morsing
145  MCULED(LEDON);
146  else
147  LED1(LEDON);
148  if ( cod & (1<<(i-1)) )
149  {
150  LOG("-");
151  delay_ms(450);
152  }
153  else
154  {
155  LOG(".");
156  delay_ms(150);
157  }
158  if (NUMLEDS == 0)
159  MCULED(LEDOFF);
160  else
161  LED1(LEDOFF);
162  delay_ms(150);
163  }
164  LOG(" %hhx\n", nibble);
165  delay_ms(450);
166 }

176 static void showError( u8 errorCode, u8 blinkState )
177 {
178  if (errorCode == 0)
179  {
180  MCULED(blinkState);
181  return;
182  }
183 
184  if (NUMLEDS < 2)
185  { // if we do not have enough LEDs to show error codes, morse them on LED1
186  morse_nibble(errorCode);
187  delay_ms(1000);
188  }
189  else
190  {
191  MCULED(errorCode & 0x01 ? blinkState : LEDOFF);
192  LED1(errorCode & 0x02 ? blinkState : LEDOFF);
193  LED2(errorCode & 0x04 ? blinkState : LEDOFF);
194  LED3(errorCode & 0x08 ? blinkState : LEDOFF);
195  }
196 }

276 int main(void)
277 {
278  u32 counter = 0;

299  LOG("\nHello World\n");
300  LOG(gLogStartup, (u8)((firmwareNumber >> 16) & 0xFF), (u8)((firmwareNumber >> 8) & 0xFF), (u8)(firmwareNumber & 0xFF));
301 
302  delay_ms(1);
303  /* Fill frequencies interval used in usb_commands.c with values */
304  /* European channels since we are developping in Europe, values can be changed using GUI */
305  Frequencies.freq[0]= 866900;
306  Frequencies.freq[1]= 865700;
307  Frequencies.freq[2]= 866300;
308  Frequencies.freq[3]= 867500;
309  Frequencies.numFreqs=4;
310 
311  // For Newton USA Profile is used. Initilize with USA Frequency
312 #if NEWTON
313  Frequencies.freq[0]= 902700;
314 #endif
315 
316  delay_ms(1);
317  readerInitStatus = as3993Initialize(Frequencies.freq[0]);
318 
319 #ifdef BUTTON
320  while (!BUTTON);
321 #endif
322 
323 
324  initCommands(); /* USB report commands */
325 #ifdef TUNER
326  tunerInit(&mainTuner);
327 #if RADON
328  tunerInit(&ant1Tuner);
329 #endif
330 #endif
331 
332  showError(readerInitStatus, ledBlinkState);
333 
334  LOG("initialize state: %x\n", readerInitStatus);
335 
336 #if ISO6B
337  iso6bInitialize();
338 #endif
339 
340  StreamDispatcherInit(SYSCLK);
341 #if !USE_UART_STREAM_DRIVER
342  USBDeviceInit(); //usb_device.c. Initializes USB module SFRs and firmware
343  USBDeviceAttach();
344 #endif
345 
346  while (1)
347  {
348  if (counter)
349  {
350  counter--;
351  }
352  else
353  {
354  counter = (10000ULL<<(readerInitStatus?3:0));
355  ledBlinkState = (~ledBlinkState) & 0x01;
356  showError(readerInitStatus, ledBlinkState);
357  if (readerInitStatus)
358  { /* If MCU is powered before AS3993 has power it is best to try to init again */
359  readerInitStatus = as3993Initialize(Frequencies.freq[0]);
360  }
361  }
362  ProcessIO(); /* main trigger for operation commands. */
363 
364  if (doCyclicInventory()) /* do cyclic inventory if necessary.*/
365  { /* if it was performed, then update blink state */
366  ledBlinkState = (~ledBlinkState) & 0x01;
367  showError(readerInitStatus, ledBlinkState);
368  }
369 #if !USE_UART_STREAM_DRIVER
370 #ifdef BUTTON
371  if (! BUTTON)
372  { /* If button is pressed then reset */
373  USBDeviceDetach();
374  USBSoftDetach();
375  delay_ms(150);
376  return 0; /* Returning from main will reset */
377  }
378 #endif
379 #endif
380  }

464 u8 cmdReadReg(u16 rxSize, const u8 * rxData, u16 * txSize, u8 * txData)
465 {
466  u8 status = ERR_PARAM;
467  //LOG("cmdReadReg: cmd:%hhx, rxsize:%hhx, txsize:%hhx\n", rxData[0], rxSize, *txSize);
468  if (rxSize < READ_REG_RX_SIZE)
469  return status;
470  switch (rxData[0])
471  {
472  case 0: //read all regs
473  status = readRegisters(txSize, txData);
474  break;
475  case 1: // read one reg
476  status = readRegister(rxData[1], txSize, txData);
477  break;
478  default:
479  status = ERR_PARAM;
480  }
481  return status;
482 }
