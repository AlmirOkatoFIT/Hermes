122 s8 iso6bInventoryRound (Tag* tags, u8 maxtags, u8 mask, u8* filter, u8 startaddress)
123 {
124  u8 found = 0;
125  u8 cntFifoLength = 0;
126  u16 resp = 0;
127  u8 uid[8];
128  u8 mem[8];
129  u8 cnt;
130  u8 max_depth = 10;
131  u8 max_commands = 50;
132  iso6bCmd nextCmd;
133 
134  /* ISO 6B Type B UID 64 Byte */
135  as3993SingleWrite(AS3993_REG_RXLENGTHUP, 0);
136  as3993SingleWrite(AS3993_REG_RXLENGTHLOW, 0x50);
137 
138  as3993EnterDirectMode();
139  DIRECT_MODE_ENABLE_SENDER() ;
140  iso6bResync();
141  delay_us(600);
142  iso6bSendInitialize();
143  delay_us(600);
144  iso6bSendSelect(mask, filter, startaddress);
145  delay_us(20);
146  DIRECT_MODE_ENABLE_RECEIVER() ;
147  as3993ExitDirectMode();
148  as3993SingleCommand(AS3993_CMD_BLOCKRX);
149  as3993SingleCommand(AS3993_CMD_ENABLERX);
150  as3993WaitForResponse(RESP_RXDONE_OR_ERROR);
151  resp = as3993GetResponse();
152  as3993ClrResponse();
153 
154  if (resp & RESP_NORESINTERRUPT )
155  {
156  return 0;
157  }
158  if ((resp & AS3993_IRQ1_RX ) && !(resp & AS3993_IRQ1_RXERR)) // one Tag found
159  {
160  cntFifoLength = as3993SingleRead(AS3993_REG_FIFOSTATUS) & 0x1F; /*get the number of bytes */
161  if (cntFifoLength)
162  as3993FifoRead(cntFifoLength, &uid[0]);
163  else
164  return 0;
165 
166  found = 1;
167  memcpy(&tags[0].epc , &uid[0] , 8);
168  }
169  else // more than one Tag found
170  {
171  /* crc error - two or more tags responded */
172  nextCmd = iso6bSendFail;
173  cnt = 2;
174  while (maxtags && (cnt < max_depth) && max_commands--)
175  {
176  as3993EnterDirectMode();
177  DIRECT_MODE_ENABLE_SENDER();
178  nextCmd();
179  DIRECT_MODE_ENABLE_RECEIVER();
180  as3993ExitDirectMode();
181  as3993SingleCommand(AS3993_CMD_BLOCKRX);
182  as3993SingleCommand(AS3993_CMD_ENABLERX);
183  as3993WaitForResponse(RESP_RXDONE_OR_ERROR);
184  resp = as3993GetResponse();
185  as3993ClrResponse();
186 
187  if (resp & AS3993_IRQ1_RXERR) // Collision
188  {
189  nextCmd = iso6bSendFail;
190  cnt ++;
191  }
192  else if (resp & AS3993_IRQ1_RX ) // Tag found
193  {
194  cntFifoLength = as3993SingleRead(AS3993_REG_FIFOSTATUS) & 0x1F;
195  if (cntFifoLength)
196  {
197  as3993FifoRead(cntFifoLength, &uid[0]);
198  iso6bRead(uid, 0x0, mem); /* send data_read command so the tag changes its state */
199  memcpy(&tags[found].epc , &uid[0] , 8);
200  found ++;
201  maxtags --;
202  cnt --;
203  nextCmd = iso6bSendSuccess;
204  }
205  else
206  {
207  nextCmd = iso6bSendFail;
208  cnt ++;
209  }
210  }
211  else if (resp & RESP_NORESINTERRUPT ) // No Response
212  {
213  nextCmd = iso6bSendSuccess;
214  cnt --;
215  }
216  else // Default
217  {
218  if ( nextCmd != iso6bSendResend )
219  {
220  nextCmd = iso6bSendResend;
221  }
222  else
223  {
224  nextCmd = iso6bSendFail;
225  cnt ++;
226  }
227  }
228  }
229  }
230 
231  return found;
232 }

234 s8 iso6bRead(u8* uid, u8 addr, u8* buffer)
235 {
236  s8 cntSend, i;
237  u8 payload[10];
238  u8 cntFifoLength;
239  u16 crc;
240  u16 resp;
241 
242  /* ISO 6B Word data 64 Bit */
243  as3993SingleWrite(AS3993_REG_RXLENGTHUP, 0);
244  as3993SingleWrite(AS3993_REG_RXLENGTHLOW, 0x50);
245  /* Enter Direct Mode and Sender */
246  as3993EnterDirectMode();
247  DIRECT_MODE_ENABLE_SENDER() ;
248  /* Prepare and Send Data */
249  cntSend = iso6bPreparePreambleDelimiter(cmdData);
250  cntSend += iso6bPrepareByte(&cmdData[cntSend], ISO6B_READ);
251  payload[0] = ISO6B_READ;
252 
253  for (i = 0; i < 8; i++)
254  {
255  payload[i+1] = uid[i];
256  cntSend += iso6bPrepareByte(&cmdData[cntSend], payload[i+1]);
257  }
258 
259  payload[9] = addr;
260  cntSend += iso6bPrepareByte(&cmdData[cntSend], addr);
261 
262  crc = calcCrc16(payload, 10);
263  cntSend += iso6bPrepareCRC(&cmdData[cntSend], crc);
264  bbRun(cmdData, cntSend);
265  /* Receive the Data */
266  delay_us(20);
267  DIRECT_MODE_ENABLE_RECEIVER();
268  as3993ExitDirectMode();
269  as3993SingleCommand(AS3993_CMD_BLOCKRX);
270  as3993SingleCommand(AS3993_CMD_ENABLERX);
271  as3993WaitForResponse(RESP_RXDONE_OR_ERROR);
272 
273  resp = as3993GetResponse();
274  as3993ClrResponse();
275 
276  if (resp & RESP_NORESINTERRUPT )
277  {
278  return ISO6B_ERR_NOTAG;
279  }
280  else if ((resp & AS3993_IRQ1_RX ) && !(resp & AS3993_IRQ1_RXERR)) // one Tag found
281  {
282  cntFifoLength = as3993SingleRead(AS3993_REG_FIFOSTATUS) & 0x1F; /*get the number of bytes */
283  if (cntFifoLength == 8)
284  {
285  as3993FifoRead(cntFifoLength, buffer);
286  return ISO6B_ERR_NONE;
287  }
288  else
289  return ISO6B_ERR_REG_FIFO;
290  }
291 
292  return ISO6B_ERR_IRQ;
293 
294 }

354  }
355 
356  return ISO6B_ERR_IRQ;

383 static s8 iso6bPreparePreambleDelimiter(t_bbData *preData)
384 {
385  /* prepare the preamble and the delimiter1 */
386  /* preamble is 9 times 01 - see section 8.1.4.3 of ISO18000-6b doc */
387  /* delimiter 1 is 1100111010 - see section 8.1.4.4.2 of ISO18000-6b doc */
388  preData[0].bbdata = 0x55; /* 01010101 */
389  preData[1].bbdata = 0x55; /* 01010101 */
390  preData[2].bbdata = 0x73; /* 01 and 110011 */
391  preData[3].bbdata = 0xa; /* 1010 */
392  preData[0].length = 8;
393  preData[1].length = 8;
394  preData[2].length = 8;
395  preData[3].length = 4;
396  return 4;
397 }

439 static s8 iso6bPrepareByte (t_bbData *manchdata, u8 byteData)
440 {
441  manchdata[0].bbdata = 0;
442  manchdata[1].bbdata = 0;
443  manchdata[0].length = 8;
444  manchdata[1].length = 8;
445 
446  if (byteData & BIT7) manchdata[0].bbdata |= 2 << 6;
447  else manchdata[0].bbdata |= 1 << 6;
448  if (byteData & BIT6) manchdata[0].bbdata |= 2 << 4;
449  else manchdata[0].bbdata |= 1 << 4;
450  if (byteData & BIT5) manchdata[0].bbdata |= 2 << 2;
451  else manchdata[0].bbdata |= 1 << 2;
452  if (byteData & BIT4) manchdata[0].bbdata |= 2 << 0;
453  else manchdata[0].bbdata |= 1 << 0;
454  if (byteData & BIT3) manchdata[1].bbdata |= 2 << 6;
455  else manchdata[1].bbdata |= 1 << 6;
456  if (byteData & BIT2) manchdata[1].bbdata |= 2 << 4;
457  else manchdata[1].bbdata |= 1 << 4;
458  if (byteData & BIT1) manchdata[1].bbdata |= 2 << 2;
459  else manchdata[1].bbdata |= 1 << 2;
460  if (byteData & BIT0) manchdata[1].bbdata |= 2 << 0;
461  else manchdata[1].bbdata |= 1 << 0;
462  return 2;
463 }

482 static s8 iso6bPrepareCRC(t_bbData *crcData, s16 crc)
483 {
484  /* send inverted crc (ISO18000-6 spec, chapter 6.5.7.3) */
485  crc = ~crc;
486  iso6bPrepareByte (crcData, (0xff00 & crc) >> 8);
487  crcData += 2;
488  iso6bPrepareByte (crcData, 0xff & crc);
489  return 4; /* return number of used bytes */
490 }

597 static void iso6bSendSelect(u8 mask, u8* filter, u8 startaddress)
598 {
599  u8 bsData[11];
600  s8 cnt;
601  s8 i;
602  u16 crc;
603 
604  cnt = iso6bPreparePreambleDelimiter(cmdData);
605  /* issue GROUP_SELECT_EQ command (0x0) */
606  bsData[0] = ISO6B_GROUP_SELECT_EQ;
607  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[0]);
608  /* address */
609  bsData[1] = startaddress;
610  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[1]);
611  /* mask - if mask is 0 all tags are selected */
612  bsData[2] = mask;
613  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[2]);
614  /* data */
615  for (i = 3; i < 11; i++)
616  {
617  bsData[i] = *filter++;
618  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[i]);
619  }
620 
621  crc = calcCrc16(bsData, 11);
622  cnt += iso6bPrepareCRC(&cmdData[cnt], crc);
623  bbRun(cmdData, cnt);
624 
625 }
